else if (currentHerb === "celery") targetPage = "%E4%B8%AD%E6%96%87/%E8%8A%B9%E8%8F%9C"; else if (currentHerb === "lemongrass") targetPage = "%E4%B8%AD%E6%96%87/%E9%A6%99%E8%8C%85"; } else if (currentLang === "malay") { if (currentHerb === "home") targetPage = "bahasa-melayu"; else if (currentHerb === "celery") targetPage = "bahasa-melayu/saderi-cina"; else if (currentHerb === "lemongrass") targetPage = "bahasa-melayu/serai-lemongrass"; } else if (currentLang === "tamil") { if (currentHerb === "home") targetPage = "%E0%AE%A4%E0%AE%AE%E0%AE%B4"; else if (currentHerb === "laksa") targetPage = "%E0%AE%A4%E0%AE%AE%E0%AE%B4/%E0%AE%85%E0%AE%A9%E0%AE%9A%E0%AE%9A-%E0%AE%87%E0%AE%B2laksa-leaf"; else if (currentHerb === "pandan") targetPage = "%E0%AE%A4%E0%AE%AE%E0%AE%B4/%E0%AE%AA%E0%AE%A3%E0%AE%9F%E0%AE%A9-%E0%AE%87%E0%AE%B2pandan-leaf"; else if (currentHerb === "wolfberry") targetPage = "%E0%AE%A4%E0%AE%AE%E0%AE%B4/%E0%AE%95%E0%AE%9C-%E0%AE%87%E0%AE%B2wolfberry"; else if (currentHerb === "celery") targetPage = "%E0%AE%A4%E0%AE%AE%E0%AE%B4/%E0%AE%9A%E0%AE%A9-%E0%AE%9A%E0%AE%B2%E0%AE%B0chinese-celery"; else if (currentHerb === "lemongrass") targetPage = "%E0%AE%A4%E0%AE%AE%E0%AE%B4/%E0%AE%87%E0%AE%9E%E0%AE%9A-%E0%AE%AA%E0%AE%B2lemongrass"; } // 3. EXECUTE REDIRECT IN THE SAME PRESENTATION TAB window.open(baseUrl + targetPage, "Library Display Tab"); // Optional: Updates a status element on your scanner page if you have one const statusEl = document.getElementById('status'); if (statusEl) { statusEl.innerText = `Current Page: ${currentHerb} (${currentLang})`; } }
